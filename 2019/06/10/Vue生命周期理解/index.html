Vue生命周期理解

发表于 2019-06-10 | 分类于 Vue 
// 组件的生命周期分为三个阶段： 初始化、运行中、销毁
//当组件从要实例化开始，到出现在页面中，经历的就是初始化阶段
//组件在使用的过程中，就是运行中阶段
//组件从准备销毁到销毁之后，经历的就是销毁阶段

//在生命周期中有多特殊的时间点，有一些函数会在对应的这些时间点自动的执行，这些函数就叫钩子函数

Vue.component('my-component', {
    template: '#component',
    data () {
        return {
            message: 'Hello Vue.js',
            num: 0
        }
    },
    methods: {
        destroy () {
            this.$destroy()
        }
    },
    //当使用组件的时候，会马上的去实例化组件来得到一个组件的实例，然后去给我们这个实例绑定一些事件监听和初始化生命周期
    //然后就会执行beforeCreate，
    beforeCreate  () {
        //特点： 没有什么用，因为在这个时候，实例的数据还没有挂载，dom更没有渲染
        console.log ('beforeCreate', this.message)
    },
    //然后为这个实例挂载数据，绑定数据监听等等之类的时候
    //再然后执行created 
    created () {
        //特点：数据已经挂载了, 但是还没有dom, 而且在这里更改数据不会触发updated函数的执行，也就是说，不会重新更新dom
        //作用： 可以做一些初始数据的获取、更新
        console.log('created', this.message, document.querySelector('h2'))
        this.message += '!!!!'
    },
    //插件实例的模板，将其编译成虚拟dom结构，放入到render函数中等待渲染
    //然后执行beforeMount
    beforeMount () {
        //特点： 和created一样 ，作用也一样
        this.message += '?????'
        console.log('beforeMount')

        this.timer = setInterval(() => {
            console.log('gogo')
            this.num++
        }, 500)

    },
    //组件实例的虚拟dom被渲染在页面中，其身上也会挂载$el属性...
    //执行mounted
    mounted () {
        // 特点：能访问到数据， 也能操作真实dom，从此时开始，再去更改数据的话就会触发updated 了
        //在这里就可以操作真实dom了，比如实例化一些插件等等...
        console.log('mounted', this.message, document.querySelector('h2'))
    },
    //在此，进入到运行中阶段，当数据被更改的时候，会马上执行beforeUpdate
    beforeUpdate () {
        // 使用到的就是更改后的数据, 但是，dom还没有更新
        // 没有什么用, 不能更改数据，会死循环
        console.log('beforeUpdate', this.message, document.querySelector('h2').innerHTML)
    },
    //创建新的虚拟dom.....，重新渲染
    //然后执行 updated
    updated () {
        // 依然不能改数据，但是dom已经，更新完成了
        // 所以，我们如果想要在数据更改之后，操作更改之后的dom，就在这里做
        console.log('updated', document.querySelector('h2').innerHTML)
    },
    //当我们要销毁组件的时候， 比如，切换组件、路由跳转都会销毁组件
    //也就是说，在这个时候组件的$destroy方法会被调用，准备开始销毁组件
    //此时就会执行beforeDestroy，进入到销毁阶段了
    beforeDestroy () {
        // 做一些善后的工作
        clearInterval(this.timer)
        console.log(this.message)
    },
    //然后取消事件、数据监听、绑定，只留下组件的dom结构,
    //然后执行destroyed
    destroyed () {
        //作用： 和beforeDestroy一样
    }
})
